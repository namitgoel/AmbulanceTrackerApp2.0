create table student(roll_no int NOT NULL primary key, name varchar(255), email varchar(255), password varchar(255), mobile bigint);

create table driver(driver_id int NOT NULL primary key, name varchar(255), password varchar(255), mobile bigint);

create table faculty(faculty_id int NOT NULL primary key, password varchar(255));

create table admin(admin_id int NOT NULL primary key, password varchar(255));

create table records(date date, start_time time, end_time time, distance_travelled int, patient_roll int NOT NULL, acommopanier1_roll int, acommopanier2_roll int, driver_id int, FOREIGN KEY (patient_roll) REFERENCES student(roll_no),FOREIGN KEY (acommopanier1_roll) REFERENCES student(roll_no),FOREIGN KEY (acommopanier2_roll) REFERENCES student(roll_no), FOREIGN KEY (driver_id) REFERENCES driver(driver_id));
